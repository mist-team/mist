58  static void compute( const T &r, const T &g, const T &b, T &R, T &G, T &B )
59  {
60  T sum = r + g + b;

64  R = r / sum;
65  G = g / sum;
66  B = b / sum;
67  }

79 
80  R = static_cast< T >( r * 255.0 / sum + 0.5 );
81  G = static_cast< T >( g * 255.0 / sum + 0.5 );
82  B = static_cast< T >( b * 255.0 / sum + 0.5 );
83  }
84  };
