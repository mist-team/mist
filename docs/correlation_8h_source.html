51 template < typename T >
52 inline double cross_correlation( const array1< T > &signal1, const array1< T > &signal2, int tau, bool biased = true )
53 {
54 
55  double result = 1.0;
56 
57  // signal size is aligned to shorter one
58  size_t signal_size = ( signal1.size( ) < signal2.size( ) ? signal1.size( ) : signal2.size( ) );
59 
60  size_t length = signal_size - tau;
61 
62  for( size_t i = 0 ; i < length ; i++ )
63  {
64  result += signal1[ i ] * signal2[ i + tau ];
65  }
66 
67  if( biased )
68  {
69  result /= static_cast< double >( signal_size );
70  }
71  else
72  {
73  result /= static_cast< double >( length );
74  }
75 
76  return( result );
77 }
