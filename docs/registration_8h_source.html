92  {
93  double v_ = 1.0 - v;
94  double v2 = v * v;
95  double v3 = v2 * v;
96  f0 = v_ * v_ * v_ / 6.0;
97  f1 = 0.5 * v3 - v2 + 4.0 / 6.0;
98  f2 = ( v2 - v3 + v ) * 0.5 + 1.0 / 6.0;
99  f3 = v3 / 6.0;
100  }

723  yy *= _1_ay;
724  zz *= _1_az;
